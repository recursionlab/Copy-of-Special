import os
import glob

# Setup paths for your sub-orthogonal directories
base_dir = r"D:\STRUCTUREDONLYMETA\StructuredOnlyMeta"
subspace_logic_dir = os.path.join(base_dir, "Subspace_Logic_Core")
subspace_syntax_dir = os.path.join(base_dir, "Subspace_Syntax_Mesh")

os.makedirs(subspace_logic_dir, exist_ok=True)
os.makedirs(subspace_syntax_dir, exist_ok=True)

# Gather files
files = glob.glob(os.path.join(base_dir, "*.md"))

print(f"[SUBSPACE_ENG] Factorizing {len(files)} files into isolated sub-dimensional spaces...")

for file_path in files:
    file_name = os.path.basename(file_path)
    
    with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
        content = f.read()
        
    # Factorization Splitter: Look for technical density markers
    logic_blocks = []
    syntax_blocks = []
    
    for line in content.split("\n"):
        # If line contains high-density math or structural code terms, route to Logic Subspace
        if any(term in line.lower() for term in ["invariant", "axiom", "cohomology", "$", "h^1"]):
            logic_blocks.append(line)
        # Otherwise, route to the Syntax/Contextual Subspace
        elif len(line.strip()) > 10:
            syntax_blocks.append(line)
            
    # Save isolated Subspace Alpha (Pure Logic)
    if logic_blocks:
        with open(os.path.join(subspace_logic_dir, f"alpha_{file_name}"), "w", encoding="utf-8") as out_l:
            out_l.write("\n".join(logic_blocks))
            
    # Save isolated Subspace Beta (Syntax/Formatting)
    if syntax_blocks:
        with open(os.path.join(subspace_syntax_dir, f"beta_{file_name}"), "w", encoding="utf-8") as out_s:
            out_s.write("\n".join(syntax_blocks))

print("[SUBSPACE_ENG] Factorization matrix complete. Zero cross-contamination achieved.")
