Copy this into a file called `reasoning_bot.py`:

```python
python# reasoning_bot.py

rules = []
facts = set()

def parse_rule(text):
    text = text.lower().strip()
    if "then" not in text or "if" not in text:
        print("Rule must be in the form: if A and B then C")
        return
    condition, conclusion = text.replace("if", "").split("then")
    cond_parts = [x.strip() for x in condition.strip().split("and")]
    conclusion = conclusion.strip()
    rules.append((set(cond_parts), conclusion))
    print(f"✅ Rule added: IF {cond_parts} THEN {conclusion}")

def add_fact(text):
    fact = text.lower().strip()
    facts.add(fact)
    print(f"✅ Fact added: {fact}")

def apply_rules():
    added = True
    while added:
        added = False
        for conditions, conclusion in rules:
            if conditions.issubset(facts) and conclusion not in facts:
                facts.add(conclusion)
                print(f"🧠 Inferred: {conclusion}")
                added = True

def ask_about(fact):
    apply_rules()
    if fact.lower().strip() in facts:
        print("✅ Yes, that is true.")
    else:
        print("❌ No, that cannot be concluded.")

def main():
    print("🤖 Reasoning Chatbot")
    print("Commands: add rule: ..., add fact: ..., ask: ..., show facts, quit")

    while True:
        user = input(">> ").strip()
        if user.startswith("add rule:"):
            parse_rule(user[len("add rule:"):])
        elif user.startswith("add fact:"):
            add_fact(user[len("add fact:"):])
        elif user.startswith("ask:"):
            ask_about(user[len("ask:"):])
        elif user == "show facts":
            print("📚 Current facts:", facts)
        elif user == "quit":
            print("👋 Goodbye!")
            break
        else:
            print("❓ I don't understand. Try again.")

if __name__ == "__main__":
    main()
```
